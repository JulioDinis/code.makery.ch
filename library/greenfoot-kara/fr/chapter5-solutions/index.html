SOLUTION TASK 5.01: Baking a Cake

public class MyKaraIO extends KaraIO {

    public void act() {
        drawRectangle(21, 4);

        stop();
    }

    public void drawRectangle(int width, int height) {
        int i = 0;
        while (i < height) {
            putLeafs(width);
            turnAround();
            multiMove(width);

            // Go to next line.
            turnRight();
            move();
            turnRight();

            i = i + 1;
        }
    }

    public void putLeafs(int count) {
        int i = 0;
        while (i < count) {
            putLeaf();
            move();
            i = i + 1;
        }
    }

    public void multiMove(int steps) {
        int i = 0;
        while (i < steps) {
            move();
            i = i + 1;
        }
    }

    public void turnAround() {
        turnLeft();
        turnLeft();
    }
}
